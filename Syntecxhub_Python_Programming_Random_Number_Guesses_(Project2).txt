import random

difficulty_choice = input("Choose difficulty (Easy, Medium, Difficult): ").strip().lower()

# Difficulty settings with range and turn limits
difficulty_settings = {
    "easy": {"range": (1, 10), "turns": 15},
    "medium": {"range": (1, 50), "turns": 10},
    "difficult": {"range": (1, 99), "turns": 5}  
} #VALID

#INVALID
if difficulty_choice not in difficulty_settings:
    raise Exception("Invalid difficulty level.")

range_low, range_high = difficulty_settings[difficulty_choice]["range"]
max_turns = difficulty_settings[difficulty_choice]["turns"]
target_number = random.randint(range_low, range_high)

#haha GAMES STARTEDDD
print(f"Number randomly selected (between {range_low}-{range_high}), good luck guessing it!")

turns = 0
while True:
    guess = int(input("Enter your guess: "))
    if guess == target_number:
        print("Guessed right, congrats! :)")
        turns = 1
        break

    choice = input("Do you wish to carry on? (Y/N): ").strip().lower()
    if choice in ["yes", "y", "yeah", "sure"]:
        pass
    elif choice in ["n", "no", "nah"]:
        break

    if guess < target_number:
        print("Guess higher")
        turns += 1
    elif guess > target_number:
        print("Guess lower")
        turns += 1

print("You guessed the number", target_number, "after", turns, "turns")
